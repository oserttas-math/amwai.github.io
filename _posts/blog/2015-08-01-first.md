{% highlight java %}
int binarySearch (int [ ] E, int first, int last, int k)
    {
    	if (last < first
    		return -1;
    	else {
    		int mid = (first + last) / 2;
    		if (k == E[mid]) return mid;
    		else if (k < E[mid])
    			return binarySearch (E, first, mid - 1, k);
    		else
    			return binarySearch (E, mid + 1, last, k);
    	}
    }
{% endhighlight %}
